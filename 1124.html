import yfinance as yf

# --- 1. å®šç¾©æ•¸æ“šä¸‹è¼‰èˆ‡æ•´ç†å‡½æ•¸ ---
def fetch_and_prepare_data(ticker_symbol):
    print(f"ğŸ“¥ æ­£åœ¨ä¸‹è¼‰ {ticker_symbol} çš„æ•¸æ“š...")
    try:
        stock = yf.Ticker(ticker_symbol)
        
        # A. å–å¾—è‚¡åƒ¹è³‡æ–™ (Daily)
        df_price = stock.history(period="5y") # è¨­å®šä¸‹è¼‰éå» 5 å¹´
        
        if df_price.empty:
            print(f"âŒ éŒ¯èª¤ï¼šæ‰¾ä¸åˆ° {ticker_symbol} çš„è‚¡åƒ¹è³‡æ–™ï¼Œè«‹æª¢æŸ¥ä»£ç¢¼æ˜¯å¦æ­£ç¢ºã€‚")
            return None

        df_price = df_price[['Close']] # åªä¿ç•™æ”¶ç›¤åƒ¹
        df_price.index = df_price.index.tz_localize(None) # ç§»é™¤æ™‚å€
        
        # B. å–å¾—è‚¡åˆ©è³‡æ–™
        dividends = stock.dividends
        dividends.index = dividends.index.tz_localize(None)
        
        # å°‡é™¤æ¯æ—¥ä½µå…¥
        df_price['dividend_amt'] = dividends
        # æ¨™è¨˜é™¤æ¯æ—¥ï¼šå¦‚æœæœ‰è‚¡åˆ©é‡‘é¡ï¼Œç•¶å¤©å°±æ˜¯é™¤æ¯æ—¥
        df_price['ex_dividend_date'] = np.where(df_price['dividend_amt'] > 0, df_price.index, pd.NaT)
        
        # C. å–å¾— EPS è³‡æ–™ (è™•ç†å­£å ±å°é½Š)
        financials = stock.quarterly_financials.T
        # å°‹æ‰¾ EPS æ¬„ä½ (åç¨±å¯èƒ½ä¸åŒï¼Œæ¨¡ç³Šæœå°‹)
        eps_col = [c for c in financials.columns if 'Basic EPS' in str(c) or 'Earnings Per Share' in str(c)]
        
        if eps_col:
            df_eps = financials[eps_col].iloc[:, 0].to_frame(name='eps') # å–ç¬¬ä¸€å€‹åŒ¹é…çš„æ¬„ä½
            df_eps.index = df_eps.index.tz_localize(None)
            
            # å°‡å­£å ± EPS åˆä½µé€²æ—¥è³‡æ–™ (ä½¿ç”¨å‘å‰å¡«å…… ffill)
            df_final = df_price.join(df_eps)
            df_final['eps'] = df_final['eps'].ffill()
        else:
            print("âš ï¸ è­¦å‘Šï¼šæ‰¾ä¸åˆ° EPS æ¬„ä½ï¼Œå°‡ä½¿ç”¨é è¨­æ¨¡æ“¬å€¼ (2.5) ä»£æ›¿ã€‚")
            df_final = df_price
            df_final['eps'] = 2.5 
            
        # å¡«è£œç¼ºå¤±çš„ EPS (é¿å…å‰›é–‹å§‹å¹¾å¤©æ˜¯ NaN)
        df_final['eps'] = df_final['eps'].fillna(method='bfill')
            
        return df_final

    except Exception as e:
        print(f"âŒ ç™¼ç”ŸéŒ¯èª¤: {e}")
        return None

def high_yield_strategy_optimized(df):
    """
    è¼¸å…¥: df (åŒ…å« 'Close', 'eps', 'ex_dividend_date')
    """
    df = df.copy()
    
    # ç¢ºä¿æ—¥æœŸç´¢å¼•æ­£ç¢º
    df['date'] = df.index
    
    # è¨ˆç®—è¿‘å››å­£ EPS ç¸½å’Œ (é€™è£¡åšç°¡å–®ä¼°ç®—ï¼šè‹¥ EPS ç‚ºå­£è³‡æ–™ï¼Œå‰‡ * 4 ç‚ºå¹´åŒ–)
    # æ³¨æ„ï¼šçœŸå¯¦æƒ…æ³å»ºè­°ç”¨ rolling sumï¼Œä½†éœ€ç¢ºä¿æ•¸æ“šé »ç‡æ­£ç¢ºã€‚
    # é€™è£¡å‡è¨­ df['eps'] å·²ç¶“æ˜¯ TTM (Trailing 12 Months) æˆ–æ˜¯æœ€è¿‘ä¸€å­£ * 4
    df['t12m_eps'] = df['eps'] * 4 
    
    # [åƒæ•¸] ä¿å®ˆé…æ¯ç‡ (é‡å° 0050/æ¬Šå€¼è‚¡)
    conservative_payout = 0.7
    
    # è¨ˆç®—é ä¼°æ•¸æ“š
    df['estimated_dividend'] = df['t12m_eps'] * conservative_payout
    df['estimated_yield'] = df['estimated_dividend'] / df['Close']
    
    # --- é€²å ´è¨Šè™Ÿ ---
    # æ¢ä»¶ï¼šä¿å®ˆé ä¼°æ®–åˆ©ç‡ > 5% ä¸” EPS > 0
    df['buy_signal'] = np.where((df['estimated_yield'] > 0.05) & (df['t12m_eps'] > 0), 1, 0)
    
    # --- å‡ºå ´è¨Šè™Ÿ (æ£„æ¯ä¿åƒ¹) ---
    # æ‰¾å‡ºæ‰€æœ‰é™¤æ¯æ—¥
    ex_dates = df[df['ex_dividend_date'].notna()].index
    
    df['sell_signal'] = 0
    # æ¨™è¨˜é™¤æ¯æ—¥çš„å‰ä¸€å¤© (T-1)
    for date in ex_dates:
        if date in df.index:
            loc = df.index.get_loc(date)
            if loc > 0:
                # åœ¨é™¤æ¯æ—¥å‰ä¸€å¤©çš„ 'sell_signal' æ¬„ä½å¡«å…¥ 1
                df.iloc[loc-1, df.columns.get_loc('sell_signal')] = 1
            
    return df
